       import { figures, figureExamples } from "@/db/schema";
       import { eq } from "drizzle-orm";
       
       export async function getFigures() {
         return db.query.figures.findMany({
           with: { examples: true },
           orderBy: (figures, { asc }) => [asc(figures.name)],
         });
       }
       
       export async function getFigureBySlug(slug: string) {
         return db.query.figures.findFirst({
           where: eq(figures.slug, slug),
           with: { examples: true },
         });
       }
       
       export async function getFiguresByCategory(category: string) {
         return db.query.figures.findMany({
           where: eq(figures.category, category),
           with: { examples: true },
         });
       }
       
       export async function searchFigures(query: string) {

           with: { examples: true },
         });

    1. Create `src/db/seed.ts` to insert data:
       ```typescript
       import { db } from "./index";
       import { figures, figureExamples } from "./schema";
       import { seedFigures } from "./seed-data";
       
       async function seed() {
         console.log("Seeding figures...");
         
         for (const figureData of seedFigures) {
           const { examples, ...figure } = figureData;
           
           const [insertedFigure] = await db.insert(figures).values(figure).returning();
           console.log(`Inserted figure: ${insertedFigure.name}`);
           
           if (examples && examples.length > 0) {
             await db.insert(figureExamples).values(
               examples.map(ex => ({
                 figureId: insertedFigure.id,
                 ...ex,
               }))
             );
             console.log(`  Inserted ${examples.length} examples`);
           }
         }
         
         console.log("Seeding completed!");
       }
       
       seed().catch(err => {
         console.error("Seeding failed:", err);
         process.exit(1);
       });
       ```
